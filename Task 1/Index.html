#include int main() { float firstNumber, secondNumber; char operation; printf("Enter first number: "); scanf("%f", &firstNumber); printf("Enter second number: "); scanf("%f", &secondNumber); printf("Choose operation (+ - * /): "); scanf(" %c", &operation); if (operation == '+') printf("Result: %.2f", firstNumber + secondNumber); else if (operation == '-') printf("Result: %.2f", firstNumber - secondNumber); else if (operation == '*') printf("Result: %.2f", firstNumber * secondNumber); else if (operation == '/') { if (secondNumber != 0) printf("Result: %.2f", firstNumber / secondNumber); else printf("Division by zero is not allowed"); } else printf("Invalid operation"); return 0; }
